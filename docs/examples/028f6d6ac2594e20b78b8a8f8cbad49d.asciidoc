resp = client.search(
    body={
        "aggs": {
            "countries": {
                "terms": {
                    "field": "artist.country",
                    "order": [
                        {"rock>playback_stats.avg": "desc"},
                        {"_count": "desc"},
                    ],
                },
                "aggs": {
                    "rock": {
                        "filter": {"term": {"genre": "rock"}},
                        "aggs": {
                            "playback_stats": {
                                "stats": {"field": "play_count"}
                            }
                        },
                    }
                },
            }
        }
    },
)
print(resp)
